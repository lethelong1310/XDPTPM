CREATE TABLE `cart` (
  `cart_id` int(11) NOT NULL AUTO_INCREMENT,
  `user_id` int(11) NOT NULL,
  `product_id` int(11) NOT NULL,
  `product_name` varchar(100) NOT NULL,
  `product_price` varchar(50) NOT NULL,
  `product_quantity` int(11) NOT NULL,
  `product_image` varchar(50) NOT NULL,
  PRIMARY KEY (`cart_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE `customer` (
  `customer_id` int(11) NOT NULL AUTO_INCREMENT,
  `customer_name` varchar(100) NOT NULL,
  `user_code` varchar(10) NOT NULL,
  `customer_phone` varchar(50) NOT NULL,
  `customer_address` varchar(200) NOT NULL,
  `customer_email` varchar(150) NOT NULL,
  `payments` int(11) NOT NULL,
  `customer_note` text DEFAULT NULL,
  `user_id` int(11) DEFAULT NULL,
  PRIMARY KEY (`customer_id`),
  KEY `user_id` (`user_id`),
  CONSTRAINT `customer_ibfk_1` FOREIGN KEY (`user_id`) REFERENCES `quanli_user` (`user_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE `donhang` (
  `order_id` int(11) NOT NULL AUTO_INCREMENT,
  `product_id` int(11) DEFAULT NULL,
  `customer_id` int(11) DEFAULT NULL,
  `mahang` int(11) DEFAULT NULL,
  `soluong` int(11) DEFAULT NULL,
  `tongDoanhThu` int(11) DEFAULT NULL,
  `trangThai` tinyint(1) DEFAULT 0,
  `ngayDatHang` date DEFAULT NULL,
  PRIMARY KEY (`order_id`),
  KEY `product_id` (`product_id`),
  KEY `customer_id` (`customer_id`),
  CONSTRAINT `donhang_ibfk_1` FOREIGN KEY (`product_id`) REFERENCES `product` (`product_id`),
  CONSTRAINT `donhang_ibfk_2` FOREIGN KEY (`customer_id`) REFERENCES `customer` (`customer_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE `giaodich` (
  `giaodich_id` int(11) NOT NULL AUTO_INCREMENT,
  `khachhang_id` int(11) DEFAULT NULL,
  `sanpham_id` int(11) DEFAULT NULL,
  `soluong` int(11) DEFAULT NULL,
  `magiaodich` varchar(50) DEFAULT NULL,
  `ngayThangNam` date DEFAULT NULL,
  PRIMARY KEY (`giaodich_id`),
  KEY `khachhang_id` (`khachhang_id`),
  KEY `sanpham_id` (`sanpham_id`),
  CONSTRAINT `giaodich_ibfk_1` FOREIGN KEY (`khachhang_id`) REFERENCES `customer` (`customer_id`),
  CONSTRAINT `giaodich_ibfk_2` FOREIGN KEY (`sanpham_id`) REFERENCES `product` (`product_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE `product` (
  `product_id` int(11) NOT NULL AUTO_INCREMENT,
  `category_id` int(11) DEFAULT NULL,
  `product_name` varchar(255) DEFAULT NULL,
  `product_details` text DEFAULT NULL,
  `product_description` text DEFAULT NULL,
  `product_price` varchar(100) DEFAULT NULL,
  `product_discount` varchar(100) DEFAULT NULL,
  `product_acitve` int(11) DEFAULT 1,
  `product_hot` int(11) DEFAULT 0,
  `product_quantity` int(11) DEFAULT NULL,
  `product_image` varchar(50) DEFAULT NULL,
  PRIMARY KEY (`product_id`),
  KEY `category_id` (`category_id`),
  CONSTRAINT `product_ibfk_1` FOREIGN KEY (`category_id`) REFERENCES `tbl_category` (`category_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;


CREATE TABLE `quanli_user` (
  `user_id` int(11) NOT NULL AUTO_INCREMENT,
  `role` tinyint(1) DEFAULT 0,
  `user_name` varchar(100) NOT NULL,
  `user_code` varchar(10) NOT NULL,
  `user_email` varchar(50) NOT NULL,
  `user_phone` varchar(10) NOT NULL,
  `user_address` varchar(255) DEFAULT NULL,
  `user_password` varchar(100) NOT NULL,
  PRIMARY KEY (`user_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE `tbl_category` (
  `category_id` int(11) NOT NULL AUTO_INCREMENT,
  `category_name` varchar(50) NOT NULL,
  PRIMARY KEY (`category_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

ALTER TABLE `cart`
  ADD CONSTRAINT `cart_ibfk_1` FOREIGN KEY (`user_id`) REFERENCES `quanli_user` (`user_id`),
  ADD CONSTRAINT `cart_ibfk_2` FOREIGN KEY (`product_id`) REFERENCES `product` (`product_id`);
